/**
 * Math library
 * Instantiated at QU.math
 * @class
 */
QU.Math = function() {
    var cos = Math.cos;
    var sin = Math.sin;
    var sqrt = Math.sqrt;
    var pi = Math.PI;
    var random = Math.random;

    /**
     * Convert degrees to radians.
     * @param d {number}- degrees
     * @returns {number}
     */
    this.rad = function(d) {

        return d * pi/180;

    };


    /**
     * Convert radians to degrees.
     * @param r {number} - radians
     * @returns {number}
     */
    this.deg = function(r) {

        return r * 180/pi;

    };


    /**
     * Return a random point on the unit sphere.
     * see: http://mathworld.wolfram.com/SpherePointPicking.html
     * @returns {THREE.Vector3}
     */
    this.randomPointOnSphere = function() {

        var theta = random() * Math.PI * 2;
        var u = random() * 2 - 1;

        return new THREE.Vector3(
            sqrt(1-u*u) * cos(theta),
            sqrt(1-u*u) * sin(theta),
            u
        );

    };


    /**
     * Rotate the vector
     * @param {THREE.Vector3} v - the vector to rotate
     * @param {THREE.Vector3} axis - the axis on which to rotate
     * @param angle - degrees to rotate (radians)
     * @param ctr {THREE.Vector3} - the center of rotation
     * @returns {THREE.Vector3}
     * @deprecated
     */
    this.rotate = function rotate(v, axis, angle, ctr) {

        consoloe.log("QU: math.rotate is deprecated, use THREE builtin.");

        var R = new THREE.Matrix3();
        R.set(  cos(angle), -sin(angle), 0,
                sin(angle), cos(angle),  0,
                0,               0,                1
        );
        v.sub(ctr);
        v.applyMatrix3(R);
        v.add(ctr);
        return v;

    };


    /**
     * Generate the vector points that ring a regular polygon.
     * @param n - the number of faces
     * @returns {THREE.Vector3[]}
     */
    this.pointsOnPoly = function pointsOnCircle(n) {

        var V = [],
            theta = (2 * Math.PI) / n;
        for (var i = 0; i<n; i++) {
            var v = QU.up.clone();
            v.applyAxisAngle(QU.forward, theta * i);
            V.push(v);
        }
        return V;

    };


};
QU.math = new QU.Math();
